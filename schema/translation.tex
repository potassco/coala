\section{ASP Facts}
In all laws except for the inertial law,
all fluents f will be replaced as follows.
\begin{itemize}
\item single fluent "f" replaced by\\
    val(f,true)
\item negated single fluent "-f" replaced by\\
    val(f,false)
\item fluent with assignment "f=identifier" replaced by\\
    val(f,identifier)
\end{itemize}
Here is a translation scheme for BC into ASP facts.
Uppercase letters (A,B,C) represent lists of fluents (/ actions in the after law),
which consist of a\_1, a\_2, a\_3, etc.  
\begin{itemize}
\item '$<$action$>$' A\\
    action(a\_1).\\
    action(a\_2).
\item '$<$fluent$>$' A\\
    fluent(a\_1).\\
    fluent(a\_2).
\item '$<$defined fluent$>$' A\\
    defined\_fluent(a\_1).\\
    defined\_fluent(a\_2).
\item a '$<$if$>$' B '$<$ifcons$>$' C\\
	static\_law(1).\\
	head(1,a).\\
	if(1,b\_1).\\
	if(1,b\_2).\\
	ifcons(1,c\_1).\\
	ifcons(1,c\_2).
\item A '$<$after$>$' B '$<$ifcons$>$' C\\
	dynamic\_law(2).\\
	head(2,a).\\
	after(2,b\_1).\\
	after(2,b\_2).\\
	ifcons(2,c\_1).\\
    ifcons(2,c\_2).
\item '$<$default$>$' f\\
	default(f).
%\item '$<$default$>$' a '$<$if$>$' C\\
%	default(3,val(f,val)).\\
%	default_if(3,c\_1).\\
%	default_if(3,c\_2).\\
%    ..
%\item '$<$default$>$' a '$<$after$>$' C\\
%	default(4,val(f,val)).\\
%	default_after(4,c\_1).\\
%	default_after(4,c\_2).\\
%    ..
\item '$<$inertial$>$' f\\
	inertial(f).
\item '$<$impossible$>$' B\\
	impossible(5,b\_1).\\
	impossible(5,b\_2).
%\item '$<$impossible$>$' B ['$<$ifcons$>$' C]\\
%	impossible(1,b\_1).\\
%	impossible(1,b\_2).\\
%	ifcons(1,c\_1).\\
%	ifcons(1,c\_2).\\
%    ..
\item '$<$nonexecutable$>$' B ['$<$if$>$' C]\\
	nonexecutable(6,b\_1).\\
	nonexecutable(6,b\_2).\\
	nonexecutable(6,c\_1).\\
	nonexecutable(6,c\_2).
%\item '$<$nonexecutable$>$' B ['$<$if$>$' C] ['$<$ifcons$>$' D]\\

%    ..
\end{itemize}
